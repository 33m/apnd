    def apple=(options = {})
      if options.respond_to?(:keys)
        apple.cert      = options[:cert]      if options[:cert]
        apple.cert_pass = options[:cert_pass] if options[:cert_pass]
        apple.host      = options[:host]      if options[:host]
        apple.port      = options[:port]      if options[:port]
      end
    end
