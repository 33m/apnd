    def run!
      EventMachine::run do
        ohai "Starting APND Daemon on #{@bind}:#{@port}"
        EventMachine::start_server(@bind, @port, APND::Daemon::Protocol) do |server|
          server.queue = @queue
        end

        EventMachine::PeriodicTimer.new(@timer) do
          process_notifications!
        end
      end
    end
